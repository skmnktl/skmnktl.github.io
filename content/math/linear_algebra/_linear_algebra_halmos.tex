\begin{prob}
   Given $\boxplus$: ($\alpha, \beta$) $\to$ $2\alpha + 2\beta$, is $\boxplus$ associative? 
\end{prob}
\begin{proof}
    We must show that $(\alpha \boxplus \beta) \boxplus \delta$ = $\alpha \boxplus (\beta \boxplus \delta)$
\end{proof}
